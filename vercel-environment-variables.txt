# VARIÁVEIS DE AMBIENTE PARA CONFIGURAR NO VERCEL
# Vá em Settings → Environment Variables no seu projeto Vercel

# === SUPABASE ===
NEXT_PUBLIC_SUPABASE_URL=sua_url_supabase_aqui
NEXT_PUBLIC_SUPABASE_ANON_KEY=sua_chave_anonima_supabase
SUPABASE_SERVICE_ROLE_KEY=sua_service_role_key_supabase

# === STRIPE ===
STRIPE_PUBLIC_KEY=pk_test_sua_chave_publica_stripe
STRIPE_SECRET_KEY=sk_test_sua_chave_secreta_stripe
STRIPE_WEBHOOK_SECRET=whsec_seu_webhook_secret_stripe

# === OPENAI ===
OPENAI_API_KEY=sk-sua_chave_openai_aqui

# === NEXTAUTH ===
NEXTAUTH_SECRET=uma_string_aleatoria_muito_segura_aqui
NEXTAUTH_URL=https://seu-projeto.vercel.app

# IMPORTANTE:
# 1. Substitua todos os valores pelos reais das suas contas
# 2. Para NEXTAUTH_SECRET, use uma string aleatória de pelo menos 32 caracteres
# 3. Para NEXTAUTH_URL, use a URL final do seu projeto no Vercel
# 4. Configure todas como "Production", "Preview" e "Development"
